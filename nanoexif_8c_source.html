00060 static inline uint32_t read_32(nanoexif_endian endian, const uint8_t *buf) {
00061     if (endian == NANOEXIF_LITTLE_ENDIAN) {
00062         return (buf[3]<<24) | (buf[2]<<16) | (buf[1]<<8) | buf[0];
00063     } else {
00064         return (buf[0]<<24) | (buf[1]<<16) | (buf[2]<<8) | buf[3];
00065     }
00066 }

00068 static inline uint16_t read_16(nanoexif_endian endian, const uint8_t *buf) {
00069     if (endian == NANOEXIF_LITTLE_ENDIAN) {
00070         return (buf[1]<<8) | buf[0];
00071     } else {
00072         return (buf[0]<<8) | buf[1];
00073     }
00074 }

00076 static inline uint16_t swap_endian_16(uint16_t i) {
00077     return ((i&0xff)<<8) | ((i&0xff00)>>8);
00078 }
00079 static inline uint32_t swap_endian_32(uint32_t i) {
00080     return ((i&0x000000ff)<<24) | ((i&0x0000ff00)<<8) | ((i&0x00ff0000)>>8) | ((i&0xff000000)>>24);

00083 static inline nanoexif * parse_app1(FILE * fp, size_t app1_len, uint32_t * ifd_offset) {
00084     if (app1_len < 6) { return NULL; }
00085 
00086     // check exif header
00087     {
00088         uint8_t exif_header[6];
00089         if (fread(exif_header, 1, 6, fp) != 6) {
00090             D("CANNOT read exif header\n");
00091             return NULL;
00092         }
00093         const char *EXIF_HEADER = "\x45\x78\x69\x66\x00\x00";
00094         if (memcmp(exif_header, EXIF_HEADER, 6)!=0) {
00095             D("EXIFHEADER\n");
00096             return NULL;
00097         }
00098     }
00099 
00100     uint8_t *buf = malloc(app1_len-6);
00101     if (!buf) { return NULL; }
00102 
00103     if (fread(buf, 1, app1_len-6, fp) != app1_len-6) {
00104         D("CANNOT read app1 header\n");
00105         free(buf);
00106         return NULL;
00107     }
00108 
00109     nanoexif_endian endian;
00110     if (memcmp(buf, "\x4d\x4d", 2) == 0) {
00111         D("BIG ENDIAN\n");
00112         endian = NANOEXIF_BIG_ENDIAN;
00113     } else { // 4949
00114         D("LITTLE ENDIAN\n");
00115         endian = NANOEXIF_LITTLE_ENDIAN;
00116     }
00117 
00118     uint16_t tag_mark = read_16(endian, buf+2);
00119     if (tag_mark == 0x2A00) {
00120         D("tiff header fail\n");
00121         free(buf);
00122         return NULL; // tiff
00123     }
00124     *ifd_offset = read_32(endian, buf+4);
00125 
00126     nanoexif * ne = malloc(sizeof(nanoexif));
00127     if (!ne) {
00128         free(buf);
00129         return NULL;
00130     }
00131     ne->endian         = endian;
00132     ne->fp             = fp;

00144 nanoexif * nanoexif_init(FILE *fp, uint32_t *ifd_offset) {
00145     {
00146         char soi[2];
00147         if (fread(soi, sizeof(char), 2, fp) != 2) {
00148             D("cannot read soi\n");
00149             return NULL;
00150         }
00151         if (soi[0] != '\xff' && soi[1] != '\xd8') {
00152             D("err, not soi");
00153             return NULL;
00154         }
00155     }
00156 
00157     /* some jpeg file put APP0 header before APP1 header. Yes, this is invalid. */
00158     while (1) {
00159         uint8_t marker_len[4];
00160         if (fread(marker_len, 1, sizeof(marker_len), fp) != sizeof(marker_len)) {
00161             D("cannot read marker\n");
00162             return NULL;
00163         }
00164         if (marker_len[0] != (uint8_t)'\xFF') {
00165             D("invalid marker\n");
00166             return NULL;
00167         }
00168 
00169         /* marker length is always big endian */
00170         uint16_t len = read_16(NANOEXIF_BIG_ENDIAN, marker_len+2);
00171 
00172         if (marker_len[1] == (uint8_t)'\xE1') { // APP1
00173             D("app1 header : %d\n", len);
00174             return parse_app1(fp, len, ifd_offset);
00175         } else if (marker_len[1] == (uint8_t)'\xDA') { // SOS
00176             /* reach to image.. hmm. this jpeg doesn't contains exif. */
00177             return NULL; /* missing exif */
00178         } else {
00179             /* skip this part... */
00180             if (fseek(fp, len-2, SEEK_CUR) != 0) {
00181                 D("cannot seek\n");
00182                 return NULL;
00183             }
00184         }
00185     }
00186     return NULL; // should not reach here

00209     *cnt = read_16(ne->endian, ne->buf + offset);
00210     nanoexif_ifd_entry * entries = malloc(sizeof(nanoexif_ifd_entry)* (*cnt));
00211     if (!entries) { return NULL; }
00212     memcpy(entries, ne->buf+offset+2, sizeof(nanoexif_ifd_entry)*(*cnt));
00213     int i;
00214     for (i=0; i<*cnt;i++) {
00215         if (NANOEXIF_MACHINE_ENDIAN != ne->endian) {
00216             entries[i].tag    = swap_endian_16(entries[i].tag);
00217             entries[i].type   = swap_endian_16(entries[i].type);
00218             entries[i].count  = swap_endian_32(entries[i].count);
00219         }

00221     uint32_t next_ifd_offset_pos = offset+2+sizeof(nanoexif_ifd_entry)*(*cnt);
