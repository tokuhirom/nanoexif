00064 static inline uint32_t read_32(nanoexif_endian endian, const uint8_t *buf) {
00065     if (endian == NANOEXIF_LITTLE_ENDIAN) {
00066         return (buf[3]<<24) | (buf[2]<<16) | (buf[1]<<8) | buf[0];
00067     } else {
00068         return (buf[0]<<24) | (buf[1]<<16) | (buf[2]<<8) | buf[3];
00069     }
00070 }

00072 static inline uint16_t read_16(nanoexif_endian endian, const uint8_t *buf) {
00073     if (endian == NANOEXIF_LITTLE_ENDIAN) {
00074         return (buf[1]<<8) | buf[0];
00075     } else {
00076         return (buf[0]<<8) | buf[1];
00077     }
00078 }

00080 static inline uint16_t swap_endian_16(uint16_t i) {
00081     return ((i&0xff)<<8) | ((i&0xff00)>>8);
00082 }
00083 static inline uint32_t swap_endian_32(uint32_t i) {
00084     return ((i&0x000000ff)<<24) | ((i&0x0000ff00)<<8) | ((i&0x00ff0000)>>8) | ((i&0xff000000)>>24);

00087 static inline nanoexif * parse_app1(FILE * fp, size_t app1_len, uint32_t * ifd_offset) {
00088     if (app1_len < 6) { return NULL; }
00089 
00090     // check exif header
00091     {
00092         uint8_t exif_header[6];
00093         if (fread(exif_header, 1, 6, fp) != 6) {
00094             D("CANNOT read exif header\n");
00095             return NULL;
00096         }
00097         const char *EXIF_HEADER = "\x45\x78\x69\x66\x00\x00";
00098         if (memcmp(exif_header, EXIF_HEADER, 6)!=0) {
00099             D("EXIFHEADER\n");
00100             return NULL;
00101         }
00102     }
00103 
00104     uint8_t *buf = malloc(app1_len-6);
00105     if (!buf) { return NULL; }
00106 
00107     if (fread(buf, 1, app1_len-6, fp) != app1_len-6) {
00108         D("CANNOT read app1 header\n");
00109         free(buf);
00110         return NULL;
00111     }
00112 
00113     nanoexif_endian endian;
00114     if (memcmp(buf, "\x4d\x4d", 2) == 0) {
00115         D("BIG ENDIAN\n");
00116         endian = NANOEXIF_BIG_ENDIAN;
00117     } else { // 4949
00118         D("LITTLE ENDIAN\n");
00119         endian = NANOEXIF_LITTLE_ENDIAN;
00120     }
00121 
00122     uint16_t tag_mark = read_16(endian, buf+2);
00123     if (tag_mark == 0x2A00) {
00124         D("tiff header fail\n");
00125         free(buf);
00126         return NULL; // tiff
00127     }
00128     *ifd_offset = read_32(endian, buf+4);
00129 
00130     nanoexif * ne = malloc(sizeof(nanoexif));
00131     if (!ne) {
00132         free(buf);
00133         return NULL;
00134     }
00135     ne->endian         = endian;
00136     ne->fp             = fp;
00137     ne->buf            = buf;

00148 nanoexif * nanoexif_init(FILE *fp, uint32_t *ifd_offset) {
00149     {
00150         char soi[2];
00151         if (fread(soi, sizeof(char), 2, fp) != 2) {
00152             D("cannot read soi\n");
00153             return NULL;
00154         }
00155         if (soi[0] != '\xff' && soi[1] != '\xd8') {
00156             D("err, not soi");
00157             return NULL;
00158         }
00159     }
00160 
00161     /* some jpeg file put APP0 header before APP1 header. Yes, this is invalid. */
00162     while (1) {
00163         uint8_t marker_len[4];
00164         if (fread(marker_len, 1, sizeof(marker_len), fp) != sizeof(marker_len)) {
00165             D("cannot read marker\n");
00166             return NULL;
00167         }
00168         if (marker_len[0] != (uint8_t)'\xFF') {
00169             D("invalid marker\n");
00170             return NULL;
00171         }
00172 
00173         /* marker length is always big endian */
00174         uint16_t len = read_16(NANOEXIF_BIG_ENDIAN, marker_len+2);
00175 
00176         if (marker_len[1] == (uint8_t)'\xE1') { // APP1
00177             D("app1 header : %d\n", len);
00178             return parse_app1(fp, len, ifd_offset);
00179         } else if (marker_len[1] == (uint8_t)'\xDA') { // SOS
00180             /* reach to image.. hmm. this jpeg doesn't contains exif. */
00181             return NULL; /* missing exif */
00182         } else {
00183             /* skip this part... */
00184             if (fseek(fp, len-2, SEEK_CUR) != 0) {
00185                 D("cannot seek\n");
00186                 return NULL;
00187             }
00188         }
00189     }
00190     return NULL; // should not reach here
00191 }
